      \if@mainmatter
        \huge\bfseries \@chapapp\space \thechapter%     was: huge
        \par\nobreak
        \vskip 17\p@%                                   was: 17pt
      \fi
    \fi
    \interlinepenalty\@M
    \LARGE \bfseries #1\par\nobreak%                    was: Huge
    \vskip 32\p@%                                       was: 40pt
  }}
